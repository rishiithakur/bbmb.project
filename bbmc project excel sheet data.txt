DAM WATER LEVEL MONITORING – PostgreSQL Database Design						
Leaflet + Django + PostgreSQL | Mobile-Friendly | 50 Dam Sites						
						
1. site_master	2. user_master	3. water_level_log	4. dam_observation	5. audit_log		
Parent table of all stations	Operator + admin accounts	Daily manual water entries	Gates/spillway/power detail	Login & change history		
						
TABLE RELATIONSHIPS						
Table	Primary Key	Foreign Keys	Relates To	Cardinality		
site_master	site_id SERIAL	—	—	Root table		
user_master	user_id SERIAL	site_id → site_master	site_master	Many users per site		
water_level_log	log_id SERIAL	site_id, user_id	site_master, user_master	Many logs per site/day		
dam_observation	obs_id SERIAL	log_id, site_id	water_level_log	One detail per log entry		
audit_log	audit_id BIGSERIAL	user_id, site_id	user_master, site_master	All actions tracked		
						
MOBILE WORKFLOW: Operator Login → Select Date → Enter Readings → Submit → Admin Verifies

-- 1. SITE MASTER	
-- ================================================	
CREATE TABLE site_master (	
  site_id               SERIAL PRIMARY KEY,	
  site_code             VARCHAR(20)   UNIQUE NOT NULL,	
  station_name          VARCHAR(150)  NOT NULL,	
  station_id_device     VARCHAR(20)   UNIQUE,	
  division              VARCHAR(100)  NOT NULL,	
  basin                 VARCHAR(100),	
  state                 VARCHAR(100)  DEFAULT 'Himachal Pradesh',	
  latitude              NUMERIC(10,6) NOT NULL,	
  longitude             NUMERIC(10,6) NOT NULL,	
  dam_type              VARCHAR(50),	
  full_reservoir_level  NUMERIC(8,3),	
  max_water_level       NUMERIC(8,3),	
  dead_storage_level    NUMERIC(8,3),	
  total_capacity_mcm    NUMERIC(12,3),	
  live_capacity_mcm     NUMERIC(12,3),	
  catchment_area_sqkm   NUMERIC(12,3),	
  instrumentation       VARCHAR(100),	
  site_status           VARCHAR(30)   DEFAULT 'Active',	
  transmission_minutes  SMALLINT,	
  mobile_number         VARCHAR(15),	
  functional            BOOLEAN       DEFAULT TRUE,	
  remarks               TEXT,	
  created_at            TIMESTAMPTZ   DEFAULT NOW(),	
  updated_at            TIMESTAMPTZ   DEFAULT NOW()	
);	
	
-- 2. USER MASTER	
-- ================================================	
CREATE TABLE user_master (	
  user_id         SERIAL PRIMARY KEY,	
  username        VARCHAR(50)   UNIQUE NOT NULL,	
  password_hash   VARCHAR(255)  NOT NULL,	
  full_name       VARCHAR(150)  NOT NULL,	
  email           VARCHAR(150)  UNIQUE,	
  mobile          VARCHAR(15),	
  role            VARCHAR(20)   DEFAULT 'operator'	
                  CHECK (role IN ('admin','operator','viewer')),	
  site_id         INTEGER       REFERENCES site_master(site_id),	
  designation     VARCHAR(100),	
  department      VARCHAR(100),	
  is_active       BOOLEAN       DEFAULT TRUE,	
  last_login      TIMESTAMPTZ,	
  login_count     INTEGER       DEFAULT 0,	
  must_change_pwd BOOLEAN       DEFAULT TRUE,	
  created_by      INTEGER       REFERENCES user_master(user_id),	
  created_at      TIMESTAMPTZ   DEFAULT NOW(),	
  updated_at      TIMESTAMPTZ   DEFAULT NOW()	
);	
	
-- 3. WATER LEVEL LOG	
-- ================================================	
CREATE TABLE water_level_log (	
  log_id               SERIAL PRIMARY KEY,	
  site_id              INTEGER       NOT NULL REFERENCES site_master(site_id),	
  user_id              INTEGER       NOT NULL REFERENCES user_master(user_id),	
  observation_date     DATE          NOT NULL,	
  observation_time     TIME          NOT NULL,	
  observation_datetime TIMESTAMPTZ   NOT NULL,	
  water_level_m        NUMERIC(8,3)  NOT NULL CHECK (water_level_m > 0),	
  water_level_ft       NUMERIC(10,3) GENERATED ALWAYS AS	
                         (water_level_m * 3.28084) STORED,	
  storage_mcm          NUMERIC(12,3),	
  storage_percent      NUMERIC(5,2),	
  inflow_cusecs        NUMERIC(12,3),	
  outflow_cusecs       NUMERIC(12,3),	
  net_change_m         NUMERIC(6,3),	
  observation_type     VARCHAR(20)   DEFAULT 'manual',	
  weather_condition    VARCHAR(50),	
  visibility           VARCHAR(30),	
  remarks              TEXT,	
  entry_source         VARCHAR(20)   DEFAULT 'mobile',	
  entry_latitude       NUMERIC(10,6),	
  entry_longitude      NUMERIC(10,6),	
  is_verified          BOOLEAN       DEFAULT FALSE,	
  verified_by          INTEGER       REFERENCES user_master(user_id),	
  verified_at          TIMESTAMPTZ,	
  is_flagged           BOOLEAN       DEFAULT FALSE,	
  flag_reason          TEXT,	
  created_at           TIMESTAMPTZ   DEFAULT NOW(),	
  updated_at           TIMESTAMPTZ   DEFAULT NOW(),	
  UNIQUE (site_id, observation_datetime)	
);	
CREATE INDEX idx_wll_site_date ON water_level_log(site_id, observation_date);	
CREATE INDEX idx_wll_obs_date  ON water_level_log(observation_date);	
	
-- 4. DAM OBSERVATION	
-- ================================================	
CREATE TABLE dam_observation (

                              REFERENCES water_level_log(log_id),	
  site_id                   INTEGER       NOT NULL

  spillway_gates_total      SMALLINT,	
  spillway_gates_open       SMALLINT,	
  spillway_opening_m        NUMERIC(6,3),	
  spillway_discharge_cusecs NUMERIC(12,3),	
  sluice_gates_total        SMALLINT,	
  sluice_gates_open         SMALLINT,	
  sluice_discharge_cusecs   NUMERIC(12,3),	
  power_units_running       SMALLINT,	
  power_generation_mw       NUMERIC(8,3),	
  power_discharge_cusecs    NUMERIC(12,3),	
  rainfall_today_mm         NUMERIC(7,2),	
  rainfall_yesterday_mm     NUMERIC(7,2),	
  rainfall_this_month_mm    NUMERIC(8,2),	
  rainfall_this_year_mm     NUMERIC(9,2),	
  upstream_level_m          NUMERIC(8,3),	
  downstream_level_m        NUMERIC(8,3),	
  tail_water_level_m        NUMERIC(8,3),	
  air_temp_celsius          NUMERIC(5,2),	
  water_temp_celsius        NUMERIC(5,2),	
  wind_speed_kmh            NUMERIC(6,2),	
  wind_direction            VARCHAR(10),	
  humidity_percent          NUMERIC(5,2),	
  seepage_lt_per_min        NUMERIC(8,3),	
  piezometer_reading        NUMERIC(8,3),	
  dam_condition             VARCHAR(30)   DEFAULT 'Normal',

  created_at                TIMESTAMPTZ   DEFAULT NOW()

-- 5. AUDIT LOG	
-- ================================================	
CREATE TABLE audit_log (	
  audit_id      BIGSERIAL PRIMARY KEY,	
  user_id       INTEGER     REFERENCES user_master(user_id),	
  site_id       INTEGER     REFERENCES site_master(site_id),	
  action        VARCHAR(50) NOT NULL,	
  table_name    VARCHAR(50),	
  record_id     INTEGER,	
  old_data      JSONB,	
  new_data      JSONB,	
  ip_address    INET,	
  device_type   VARCHAR(20),	
  browser       VARCHAR(100),	
  gps_latitude  NUMERIC(10,6),	
  gps_longitude NUMERIC(10,6),	
  status        VARCHAR(20) DEFAULT 'success',	
  error_message TEXT,	
  session_id    VARCHAR(100),	
  created_at    TIMESTAMPTZ DEFAULT NOW()	
);	
CREATE INDEX idx_audit_user    ON audit_log(user_id);	
CREATE INDEX idx_audit_site    ON audit_log(site_id);	
CREATE INDEX idx_audit_created ON audit_log(created_at);	
	
-- ================================================	
-- AUTO-UPDATE trigger for updated_at columns	
-- ================================================	
CREATE OR REPLACE FUNCTION update_timestamp()	
RETURNS TRIGGER AS $$	
BEGIN	
  NEW.updated_at = NOW();	
  RETURN NEW;	
END;	
$$ LANGUAGE plpgsql;	
	
CREATE TRIGGER trg_site_master_upd	
  BEFORE UPDATE ON site_master	
  FOR EACH ROW EXECUTE FUNCTION update_timestamp();	
	
CREATE TRIGGER trg_user_master_upd	
  BEFORE UPDATE ON user_master	
  FOR EACH ROW EXECUTE FUNCTION update_timestamp();	
	
CREATE TRIGGER trg_water_level_log_upd	
  BEFORE UPDATE ON water_level_log	
  FOR EACH ROW EXECUTE FUNCTION update_timestamp();
